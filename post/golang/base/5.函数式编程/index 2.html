    package main

func adder() func(int) int {
    sum := 0 // 自由变量
    return func (v int) int {
        sum += v //v 局部变量
        return sum
    }
}

func main() {
    a := adder()
    for i := 0; i < 10; i++ {
        fmt.Printf("0 + 1 + ... + %d = %d\n", i, a(i))
    }
}

  所以，当函数返回的时候，返回的是一个闭包。我们return这个func不是说返回这个代码，而是返回了这个函数以及对sum的引用

    package main

// 1, 1, 2, 3, 5, 8, 13, ...
//    a, b
//       a  b
func fibonacci func() int {
    a, b := 0, 1 //自由变量
    return func() int {
        a, b = b, a + b
        return a
    }
}

func main() {
    f := fibonacci() //生成器
    fmt.Println(f()) //1
    fmt.Println(f()) //1
    fmt.Println(f()) //2
    fmt.Println(f()) //3
    fmt.Println(f()) //5
    fmt.Println(f()) //8
    fmt.Println(f()) //13
    fmt.Println(f()) //21
}

    type Node struct {
    value int
    Left, Right *Node
}

func (node *Node) TraverseFunc(f func(*Node)) {
    if node == nil {
        return
    }
    node.Left.TraverseFunc(f)
    f(node)
    node.Right.TraverseFunc(f)
}

func (node *Node) Traverse() {
    node.TraverseFunc(func(n *Node){
        n.Print()
    })
    fmt.Println()
}

func (node *Node) Print() {
    fmt.Println(node.value, " ")
}
